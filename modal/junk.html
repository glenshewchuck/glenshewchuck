IFRAME Test

Close the modal dialog from a link inside the IFRAME:

$ = parent.$;
window.onload = function () {
        var iframe = $('.simplemodal-data iframe').contents().find('body');
        $('.simplemodal-close', iframe).bind('click', function () {
                parent.dialog.modal('close');
        });
}

<a href='#' class='simplemodal-close'>Close</a>
